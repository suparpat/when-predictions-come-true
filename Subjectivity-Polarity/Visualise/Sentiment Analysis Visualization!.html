hour = []
hourly_sentiments = []
count = len(DateListSentiment)

temp = []

for t in range(1,count):
    if DateListSentiment[t-1].hour==DateListSentiment[t].hour and DateListSentiment[t-1].day==DateListSentiment[t].day and DateListSentiment[t-1].month==DateListSentiment[t].month and DateListSentiment[t-1].year==DateListSentiment[t].year:
        temp.append(sentiments.polarity[t-1])
        
    elif len(temp)!=0:
        count_temp = len(temp)
        hour.append('{} {} {} {}'.format(DateListSentiment[t-1].hour,DateListSentiment[t-1].day,DateListSentiment[t-1].month,DateListSentiment[t-1].year))
        hourly_sentiments.append(sum(temp)/count_temp)
        temp = []

combined = zip(hour,hourly_sentiments)

dfx = DataFrame(data = combined, columns = ['hour','hourly_sentiments'])
dfx.to_csv('combined.csv',index=False)
